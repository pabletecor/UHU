Cada sincronización tiene una cola posible.

this.wait()		//Pone a esperar a un hilo.
this.notify()		//Activa a UN hilo de la cola de hilo esperando
this.notifyAll()	//Activa a TODOS los hilos de la cola de hilos esperando


synchronized(A){

A.wait();
A.notify();

}


synchronized(B){


B.wait();
B.notify();

}
