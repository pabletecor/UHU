MF-2.

select nombre,direccion
FROM mf.CLIENTE
WHERE (extract (year from f_nac) = '1973' OR 
extract (year from f_nac) = '1985')
AND cp LIKE '15%'
ORDER BY nombre asc, direccion dec;

FROM mf.LLAMADA
