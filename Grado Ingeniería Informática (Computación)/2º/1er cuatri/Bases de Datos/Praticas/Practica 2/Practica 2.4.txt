FROM mf.CLIENTE c
WHERE EXISTS (SELECT *
		FROM MF.TELEFONO tf INNER JOIN MF.LLAMADA ll ON ll.tf_origen = tf.numero WHERE tf.CLIENTE = c.DNI AND TO_CHAR (ll.FECHA_HORA, 'dd/mm/yyyy' = '16/10/06')

FROM mf.cliente c
WHERE EXISTS (SELECT *
		FROM mf.teledono tf INNER JOIN MF.LLAMADA ll ON ll.tf_origen = tf.numero WHERE tf.CLIENTE = c.DNI AND ll.duracion <'90');
