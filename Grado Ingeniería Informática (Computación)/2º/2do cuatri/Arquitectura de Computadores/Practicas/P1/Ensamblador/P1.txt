.MODEL small

.DATA
mensaje db ' -ABCDEFGHIJKLMNÑOPQRSTUVWXYZ-ABCDEFGHIJKLMNÑOPQRSTUVWXYZ-ABCDEFGHIJKLMNÑOPQRSTUVWXYZ-ABCDEFGHIJKLMNÑOPQRSTUVWXYZ *'
;Cadena formada por 1 comienzo de cadena + 36 char * 3 + 3 giones + 2 espacios + 1 asterisco

.CODE
	MOV AX, seg mensaje
	MOV DS,AX

	MOV AL,03h
	MOV AH,0
	INT 10h


	MOV AX, 0B800h ;dir memoria modo texto
	MOV ES, AX
pinta_cadena:
	MOV AL, mensaje[si]
	MOV AH, 0Eh
	MOV ES:[DI], AX
	ADD DI,2

	INC SI
	CMP mensaje[SI], ""
	JNE pinta_cadena

	MOV SI,0
	INC mensaje[0]
	CMP DI, 4158 ;EL VALOR MAXIMO QUE PUEDA ALCANZAR DI ES 4158
	JBE pinta_cadena
	MOV DI,0 ;si ha llegado al final vuelve al principio de la pantalla
	jmp pinta_cadena ;si ha llegado al final vuelve al principio de la pantalla

END
