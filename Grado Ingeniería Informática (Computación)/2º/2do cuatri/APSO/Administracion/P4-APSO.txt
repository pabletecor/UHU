--TIPOS SHELL--

	-sh (Shell Bourne) -> /bin/sh (cambia el bash por una cacota)
	-bash (Shell Bourne Again) -> /bin/bash
	-exit/ ctrl + D -> Volver a la bash anterior

--VARIABLES--

	-VAR=valor ->Inicializacion de la cadena = cadena
	-$VAR -> uso de la variable anteriormente creada
	-set -> consultar variables del sistema (no inicia ninguna variable)
	-unset $nombre_variable -> elimina la variable
	-set -/+ z -> on/off opciones del bash

--ALIAS--

	alias n='cmd1' -> crear alias 
	alias -> mostrar alias
	
--CONFIGURACIONES--

	/etc -> directorio configuraciones globales
	~ -> directorio configuraciones de usuario (. profile)
	variable prompt principal -> PS1
